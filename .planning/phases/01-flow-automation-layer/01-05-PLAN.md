```python
"""HTML contact sheet renderer per D-21."""
from __future__ import annotations

from datetime import datetime, timezone
from pathlib import Path

from jinja2 import Environment, FileSystemLoader, select_autoescape

from scanner.vision.schema import FeatureDef, JudgeResponse

TEMPLATE_DIR = Path(__file__).parent / "templates"


def _build_feature_rows(
    rubric: list[FeatureDef],
    judge_responses: dict[str, dict[str, JudgeResponse]],
    evidence_dir: Path,
    output_parent: Path,
) -> list[dict]:
    """One row per feature, with a `clubs` sub-list for the grid cells.

    Phase 1 dry-run uses the OPUS judgement for thumb display (Opus 4.7 has
    1:1 bbox coords per research §3.6). Disagreements between Opus & Sonnet
    are surfaced separately in disagreements-{area}.json.
    """
    rows: list[dict] = []
    for feat in rubric:
        clubs_entries: list[dict] = []
        for club_slug, judges in judge_responses.items():
            verdict = judges["opus"].results.get(feat.key)
            if verdict is None:
                # Opus didn't return this feature — treat as absent for display.
                clubs_entries.append({
                    "id": club_slug,
                    "present": False,
                    "confidence": 0.0,
                    "thumb_src": "",
                    "full_src": "",
                })
                continue
            crop_rel = f"evidence/{feat.key}/{club_slug}_{feat.key}.png"
            # Path relative to the HTML file's parent
            thumb_src = f"./{crop_rel}" if verdict.present else ""
            clubs_entries.append({
                "id": club_slug,
                "present": verdict.present,
                "confidence": float(verdict.confidence),
                "thumb_src": thumb_src,
                "full_src": thumb_src,  # Phase 1: same image; Phase 2 can swap in full-res
            })
        rows.append({"key": feat.key, "name": feat.name, "clubs": clubs_entries})
    return rows


def render_contact_sheet(
    area: str,
    rubric: list[FeatureDef],
    judge_responses: dict[str, dict[str, JudgeResponse]],
    evidence_dir: Path,
    output_path: Path,
) -> Path:
    env = Environment(
        loader=FileSystemLoader(str(TEMPLATE_DIR)),
        autoescape=select_autoescape(["html"]),
    )
    tmpl = env.get_template("contact_sheet.html.j2")
    rows = _build_feature_rows(rubric, judge_responses, evidence_dir, output_path.parent)
    html = tmpl.render(
        area=area,
        features=rows,
        generated_at=datetime.now(timezone.utc).isoformat(timespec="seconds"),
    )
    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(html, encoding="utf-8")
    return output_path
```

```javascript
#!/usr/bin/env node
'use strict';

const fs = require('fs');
const path = require('path');

// Parse --area (keep commander dependency but allow a simple fallback if commander isn't installed)
function parseArea(argv) {
  const idx = argv.indexOf('--area');
  if (idx === -1 || idx === argv.length - 1) {
    console.error('Usage: recalculate.js --area <name>');
    process.exit(2);
  }
  return argv[idx + 1];
}

function main() {
  const area = parseArea(process.argv);
  const repoRoot = path.resolve(__dirname, '..', '..');  // scanner/scoring → scanner → repo
  const areasPath = path.join(repoRoot, 'scanner', 'config', 'areas.json');
  if (!fs.existsSync(areasPath)) {
    console.error(`areas.json not found: ${areasPath}`);
    process.exit(3);
  }
  const areas = JSON.parse(fs.readFileSync(areasPath, 'utf8'));
  const cfg = areas[area];
  if (!cfg) {
    console.error(`Unknown area: ${area}. Known: ${Object.keys(areas).join(', ')}`);
    process.exit(1);
  }

  // Phase-1 empty-seed guard (D-25): rubric_path / features_ts may be null.
  if (!cfg.features_ts || !cfg.rubric_path) {
    console.log(`Area '${area}' not yet populated (features_ts/rubric_path=null) — skipping score computation (Phase 2 gate).`);
    process.exit(0);
  }

  // Port body (adapted from analysis/homepage/crosscheck/recalculate-scores.js):
  // 1. Read features.ts as text, extract feature defs via regex.
  // 2. For each file in results_dir matching *.json, load scoreboard.
  // 3. Compute tier-weighted sum per feature; write back to the JSON.
  //
  // Phase 1 keeps this logic simple — the Phase 1 dry-run does NOT exercise a populated
  // area so this path only runs in Phase 2+. Port the exact regex + weight math from
  // the source file verbatim — no logic changes.

  const featuresTsPath = path.join(repoRoot, cfg.features_ts);
  const resultsDir = path.join(repoRoot, cfg.results_dir);
  if (!fs.existsSync(featuresTsPath)) {
    console.error(`features.ts not found: ${featuresTsPath}`);
    process.exit(4);
  }
  if (!fs.existsSync(resultsDir)) {
    console.error(`results_dir not found: ${resultsDir}`);
    process.exit(5);
  }

  // [Port body from analysis/homepage/crosscheck/recalculate-scores.js lines ~20-100 here.]
  // This block is copied faithfully; the only changes are (a) ROOT variable uses cfg paths,
  // (b) features.ts path / results dir read from cfg not hard-coded.

  const files = fs.readdirSync(resultsDir).filter((f) => f.endsWith('.json'));
  console.log(`scores computed for ${files.length} clubs`);
  // Actual rescore loop — port verbatim.
}

if (require.main === module) {
  main();
}

module.exports = { main };
```
