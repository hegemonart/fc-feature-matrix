```python
from typing import Literal
from pydantic import BaseModel, Field, field_validator

class FormField(BaseModel):
    selector: str
    value: str  # e.g. "Test Test", "test@example.com"

class FlowStep(BaseModel):
    step_name: str
    url: str | None = None
    # D-16 invariant: 'submit' deliberately excluded from the Literal union.
    action: Literal["navigate", "click", "fill", "wait", "screenshot"]
    selector: str | None = None
    form_fields: list[FormField] | None = None
    hide_selectors: list[str] = Field(default_factory=list)
    wait_for: str | None = None  # CSS selector or 'networkidle'

class FlowMap(BaseModel):
    area: str
    club: str
    entry_url: str
    steps: list[FlowStep] = Field(min_length=1, max_length=15)  # D-15
```

<!-- scanner/vision/schema.py -->
```python
from pydantic import BaseModel, Field

class FeatureDef(BaseModel):
    key: str               # snake_case
    name: str
    yes_criterion: str

class FeatureVerdict(BaseModel):
    present: bool
    step: str
    evidence_bbox: tuple[float, float, float, float] | None
    confidence: float = Field(ge=0.0, le=1.0)
    notes: str = Field(max_length=500)

class FeatureResult(BaseModel):
    feature_key: str
    verdict: FeatureVerdict

class JudgeResponse(BaseModel):
    """Output of a single judge (opus OR sonnet)."""
    model: str                          # "claude-opus-4-7" | "claude-sonnet-4-6"
    results: dict[str, FeatureVerdict]  # keyed by feature_key
```

Simplification note on `AreasConfig`: the interface sketch lists `hospitality: AreaEntry | None = None` as a named field — but since Phase 3-8 add more areas (`ticket`, `sponsorship`, etc.), **use `dict[str, AreaEntry]` as the actual top-level shape** rather than named attributes:

```python
from pydantic import RootModel

class AreasConfig(RootModel[dict[str, AreaEntry]]):
    """Top-level areas.json = {area_name: AreaEntry, ...}. Use .root for dict access."""

    def get(self, area: str) -> AreaEntry:
        try:
            return self.root[area]
        except KeyError as e:
            raise KeyError(f"Unknown area: {area}. Known: {list(self.root)}") from e
```

This keeps Phase 2+ extension painless (just add a new top-level key in `scanner/config/areas.json`). Plan 07 ships the hospitality-only seed.
