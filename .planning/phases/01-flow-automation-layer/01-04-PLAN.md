```python
# scanner/vision/factory.py
from typing import Literal
from scanner.vision.client import VisionClient

ApiMode = Literal["subscription", "api-key"]


def get_client(api_mode: ApiMode, model: str) -> VisionClient:
    """Dispatch between subscription (default, D-26a) and api-key (fallback, D-26b)."""
    if api_mode == "subscription":
        from scanner.vision.client_subscription import SubscriptionVisionClient
        return SubscriptionVisionClient(model=model)
    if api_mode == "api-key":
        import os
        key = os.environ.get("ANTHROPIC_API_KEY")
        if not key:
            raise RuntimeError(
                "ANTHROPIC_API_KEY not set — required for --api-mode api-key. "
                "Either export the key or use --api-mode subscription (default)."
            )
        from scanner.vision.client_apikey import APIKeyVisionClient
        return APIKeyVisionClient(api_key=key, model=model)
    raise ValueError(f"Unknown api_mode: {api_mode!r}")
```

```python
    image_path: Path,

    *,
    api_mode: str = "subscription",
```

```python
"""Checklist-first prompts per D-18 and OUTPUT_SCHEMA per research §3.3."""
from __future__ import annotations

import json
from scanner.vision.schema import FeatureDef

OUTPUT_SCHEMA: dict = {
    "type": "object",
    "patternProperties": {
        "^[a-z_]+$": {
            "type": "object",
            "properties": {
                "present": {"type": "boolean"},
                "step": {"type": "string"},
                "evidence_bbox": {
                    "anyOf": [
                        {"type": "null"},
                        {"type": "array", "items": {"type": "number"},
                         "minItems": 4, "maxItems": 4}
                    ]
                },
                "confidence": {"type": "number", "minimum": 0, "maximum": 1},
                "notes": {"type": "string", "maxLength": 500},
            },
            "required": ["present", "step", "evidence_bbox", "confidence", "notes"],
            "additionalProperties": False,
        }
    },
    "additionalProperties": False,
}


def build_checklist_prompt(features: list[FeatureDef], step_name: str = "landing") -> str:
    """Per D-18 (checklist-first, never open-ended). step_name labels the screenshot's flow-map step."""
    rubric_lines = "\n".join(
        f"- key: {f.key}\n  name: {f.name}\n  qualifies_as_yes_if: {f.yes_criterion}"
        for f in features
    )
    return (
        "You are auditing a web page screenshot against a frozen feature checklist.\n"
        "Do NOT identify features not on the list. For EACH feature, return a JSON\n"
        "object with: present (bool), step (string — use the provided step name),\n"
        "evidence_bbox ([x, y, w, h] in pixel coordinates of the attached screenshot; null if absent),\n"
        "confidence (0.0-1.0), notes (brief, max 500 chars).\n\n"
        f"step = \"{step_name}\"\n\n"
        "Feature checklist:\n"
        f"{rubric_lines}\n\n"
        "Return ONE JSON object keyed by feature_key. No prose."
    )
```

**scanner/vision/client_apikey.py** — follow research §3.3 pattern verbatim, adapted:

```python
"""APIKeyVisionClient — D-26(b) pay-per-token fallback via anthropic SDK."""
from __future__ import annotations

import base64
import json
import logging
from pathlib import Path

import anthropic

from scanner.vision.prompts import OUTPUT_SCHEMA, build_checklist_prompt
from scanner.vision.schema import FeatureDef, FeatureVerdict

logger = logging.getLogger(__name__)


class APIKeyVisionClient:
    """Uses anthropic.Anthropic() — requires ANTHROPIC_API_KEY."""

    def __init__(self, api_key: str, model: str):
        self.model = model
        self._client = anthropic.Anthropic(
            api_key=api_key,
            default_headers={"anthropic-beta": "structured-outputs-2025-11-13"},
        )

    def analyze_screenshot(
        self,
        image_path: Path,
        rubric: list[FeatureDef],
    ) -> dict[str, FeatureVerdict]:
        image_b64 = base64.b64encode(Path(image_path).read_bytes()).decode("utf-8")
        prompt = build_checklist_prompt(rubric)
        resp = self._client.messages.create(
            model=self.model,
            max_tokens=4096,
            output_format={"type": "json_schema", "schema": OUTPUT_SCHEMA},
            messages=[{
                "role": "user",
                "content": [
                    {"type": "image",
                     "source": {"type": "base64", "media_type": "image/png", "data": image_b64}},
                    {"type": "text", "text": prompt},
                ],
            }],
        )
        raw = json.loads(resp.content[0].text)
        return {k: FeatureVerdict(**_clamp_confidence(v)) for k, v in raw.items()}

    def ask_yes_no(self, screenshot_path: Path, prompt: str) -> str:
        image_b64 = base64.b64encode(Path(screenshot_path).read_bytes()).decode("utf-8")
        resp = self._client.messages.create(
            model=self.model,
            max_tokens=10,
            messages=[{
                "role": "user",
                "content": [
                    {"type": "image",
                     "source": {"type": "base64", "media_type": "image/png", "data": image_b64}},
                    {"type": "text", "text": prompt},
                ],
            }],
        )
        return resp.content[0].text.strip().lower()


def _clamp_confidence(verdict: dict) -> dict:
    """LLM sometimes returns 1.0001 / -0.001. Clamp before Pydantic rejects."""
    if "confidence" in verdict:
        verdict = {**verdict, "confidence": max(0.0, min(1.0, float(verdict["confidence"])))}
    return verdict
```

**scanner/vision/client_subscription.py** — analogous but uses `claude_agent_sdk`. The Claude Agent SDK exposes a `query()` function returning an async iterator of messages. For Plan 1 Phase-1 dry-run we run synchronously via `asyncio.run`:

```python
"""SubscriptionVisionClient — D-26(a) default. Uses claude-agent-sdk → Max 20x subscription.

No API key required. Requires `claude` CLI on PATH at runtime (checked by Plan 03 and Plan 08).
"""
from __future__ import annotations

import asyncio
import base64
import json
import logging
from pathlib import Path

from claude_agent_sdk import query  # type: ignore[import-untyped]

from scanner.vision.prompts import build_checklist_prompt
from scanner.vision.schema import FeatureDef, FeatureVerdict

logger = logging.getLogger(__name__)


class SubscriptionVisionClient:
    """Uses claude_agent_sdk.query — consumes Max 20x quota."""

    def __init__(self, model: str):
        self.model = model

    def analyze_screenshot(
        self,
        image_path: Path,
        rubric: list[FeatureDef],
    ) -> dict[str, FeatureVerdict]:
        image_b64 = base64.b64encode(Path(image_path).read_bytes()).decode("utf-8")
        prompt = build_checklist_prompt(rubric)
        content = [
            {"type": "image",
             "source": {"type": "base64", "media_type": "image/png", "data": image_b64}},
            {"type": "text",
             "text": prompt + "\n\nReturn ONLY valid JSON — no surrounding text, no markdown fences."},
        ]
        text = asyncio.run(self._collect_text(content=content))
        raw = json.loads(text)
        from scanner.vision.client_apikey import _clamp_confidence  # reuse clamp
        return {k: FeatureVerdict(**_clamp_confidence(v)) for k, v in raw.items()}

    def ask_yes_no(self, screenshot_path: Path, prompt: str) -> str:
        image_b64 = base64.b64encode(Path(screenshot_path).read_bytes()).decode("utf-8")
        content = [
            {"type": "image",
             "source": {"type": "base64", "media_type": "image/png", "data": image_b64}},
            {"type": "text", "text": prompt},
        ]
        text = asyncio.run(self._collect_text(content=content))
        return text.strip().lower()

    async def _collect_text(self, content: list[dict]) -> str:
        """Drain the claude-agent-sdk async iterator into a single concatenated text string."""
        out_parts: list[str] = []
        async for message in query(prompt=content, options={"model": self.model}):
            # claude_agent_sdk yields Message objects with .content blocks; extract text.
            for block in getattr(message, "content", []) or []:
                if getattr(block, "type", None) == "text":
                    out_parts.append(block.text)
        return "".join(out_parts)
```

> **Note on claude_agent_sdk API:** the exact `query()` signature may differ; the runtime shape is an async iterator of messages. If the actual SDK surface differs from what's shown, the executor may adapt the `_collect_text` implementation — but **the public interface of `SubscriptionVisionClient` (constructor, `analyze_screenshot`, `ask_yes_no`) MUST NOT change**. Downstream plans depend on those signatures.

```python
"""Disagreement detection per D-19 + research §3.4 (three-rule consensus check)."""
from __future__ import annotations

import json
from dataclasses import asdict, dataclass
from pathlib import Path

from scanner.vision.schema import FeatureVerdict


@dataclass
class Disagreement:
    feature_key: str
    kind: str  # "presence" | "confidence" | "bbox"
    opus: dict
    sonnet: dict


def bbox_iou(a: tuple, b: tuple) -> float:
    ax, ay, aw, ah = a
    bx, by, bw, bh = b
    ix1, iy1 = max(ax, bx), max(ay, by)
    ix2, iy2 = min(ax + aw, bx + bw), min(ay + ah, by + bh)
    iw, ih = max(0, ix2 - ix1), max(0, iy2 - iy1)
    inter = iw * ih
    union = aw * ah + bw * bh - inter
    return inter / union if union > 0 else 0.0


def find_disagreements(
    opus_results: dict[str, FeatureVerdict],
    sonnet_results: dict[str, FeatureVerdict],
) -> list[Disagreement]:
    out: list[Disagreement] = []
    for fkey in opus_results:
        if fkey not in sonnet_results:
            continue  # checklist mismatch — handled elsewhere
        o, s = opus_results[fkey], sonnet_results[fkey]
        o_d, s_d = o.model_dump(), s.model_dump()
        if o.present != s.present:
            out.append(Disagreement(fkey, "presence", o_d, s_d))
            continue
        if abs(o.confidence - s.confidence) > 0.3:
            out.append(Disagreement(fkey, "confidence", o_d, s_d))
        if o.present and s.present and o.evidence_bbox and s.evidence_bbox:
            if bbox_iou(o.evidence_bbox, s.evidence_bbox) < 0.5:
                out.append(Disagreement(fkey, "bbox", o_d, s_d))
    return out


def write_disagreements(disagreements: list[Disagreement], path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps([asdict(d) for d in disagreements], indent=2))
```

```python
"""PIL crop + DPR/model-scale-aware bbox denormalisation.

Ported from analysis/homepage/crosscheck/recapture_round5.py::capture_from_fullpage.
"""
from __future__ import annotations

from dataclasses import dataclass
from io import BytesIO
from pathlib import Path

from PIL import Image

MIN_CROP_PX = 50
PAD_PX = 10


@dataclass
class SliceResult:
    ok: bool
    path: Path | None = None
    reason: str | None = None


def denormalise_bbox(
    bbox: tuple[float, float, float, float],
    model: str,
    original_w: int,
    original_h: int,
) -> tuple[float, float, float, float]:
    """Scale bbox from model-coordinate-space to original-image pixel space.

    Opus 4.7: resizes input to max long-edge 2576 before analysis.
    Sonnet 4.6 / Haiku: resizes to max long-edge 1568.
    If original long edge is under the model's limit, no resize happened — bbox is 1:1.
    """
    long_edge_limit = 2576 if model.startswith("claude-opus") else 1568
    actual_long = max(original_w, original_h)
    if actual_long <= long_edge_limit:
        return bbox
    scale = actual_long / long_edge_limit
    return (bbox[0] * scale, bbox[1] * scale, bbox[2] * scale, bbox[3] * scale)


def slice_feature(
    fullpage_png: bytes,
    bbox: tuple[float, float, float, float],
    out_path: Path,
) -> SliceResult:
    img = Image.open(BytesIO(fullpage_png))
    fw, fh = img.size
    x, y, w, h = (int(v) for v in bbox)
    x1 = max(0, x - PAD_PX)
    y1 = max(0, y - PAD_PX)
    x2 = min(fw, x + w + PAD_PX)
    y2 = min(fh, y + h + PAD_PX)
    if x2 <= x1 or y2 <= y1:
        return SliceResult(ok=False, reason="bbox outside image bounds")
    cw, ch = x2 - x1, y2 - y1
    if cw < MIN_CROP_PX or ch < MIN_CROP_PX:
        return SliceResult(ok=False, reason=f"crop too small: {cw}x{ch}")
    cropped = img.crop((x1, y1, x2, y2))
    out_path.parent.mkdir(parents=True, exist_ok=True)
    cropped.save(out_path, format="PNG", optimize=True)
    return SliceResult(ok=True, path=out_path)
```
