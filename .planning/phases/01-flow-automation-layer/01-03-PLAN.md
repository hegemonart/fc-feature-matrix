```python
"""Post-capture cookie/consent banner verification (D-14) via Haiku for speed."""
from __future__ import annotations

import logging
from pathlib import Path

logger = logging.getLogger(__name__)


def verify_banner_gone(screenshot_path: Path, *, api_mode: str = "subscription") -> bool:
    """Ask Claude Haiku 'is any cookie/consent/newsletter banner visible?' yes/no.

    Returns True if NO banner visible (capture clean). False if a banner is still
    there; caller should re-try dismissal with stronger handling.

    Uses the dual-backend factory from Plan 04 (lazy import) but pins model to Haiku
    for minimal token cost — this is a binary yes/no, not a feature detection.
    """
    try:
        from scanner.vision.factory import get_client  # lazy — avoids circular import
    except ImportError:
        logger.warning(
            "scanner.vision.factory not importable yet — banner check skipped (assumes clean). "
            "This is expected in Plan 03's isolated test runs; Plan 04 ships the factory."
        )
        return True

    client = get_client(api_mode=api_mode, model="claude-haiku-4-5")
    prompt = (
        "Is there any cookie-consent banner, newsletter signup popup, or modal dialog "
        "visible in this screenshot? Answer with exactly ONE WORD: 'yes' or 'no'."
    )
    # Uses the client's single-turn raw prompt path — implementation-defined by Plan 04.
    # For Plan 03's mocked test, we pass through client.ask_yes_no(screenshot_path, prompt).
    response: str = client.ask_yes_no(screenshot_path=screenshot_path, prompt=prompt)
    return response.strip().lower() == "no"
```

```python
"""Full-page capture orchestrator — the one function Plan 07's CLI and Plan 08's dry-run call."""
from __future__ import annotations

import logging
from pathlib import Path

from scanner.capture.browser import create_browser, scroll_lazy
from scanner.capture.cookies import dismiss_cookies

logger = logging.getLogger(__name__)


def capture_page(
    url: str,
    club: str,
    area: str,
    step_name: str,
    output_dir: Path,
    *,
    headless: bool = False,
    hide_selectors: list[str] | None = None,
) -> Path:
    """Navigate, dismiss cookies, (optionally) hide selectors, scroll, screenshot full page.

    Writes to output_dir / 'fullpage' / f'{club}_{step_name}.png' (user decision 1:
    scanner/output/evidence/{area}/fullpage/, NEVER analysis/).
    """
    out_file = output_dir / "fullpage" / f"{club}_{step_name}.png"
    out_file.parent.mkdir(parents=True, exist_ok=True)

    pw, ctx = create_browser(club=club, area=area, headless=headless)
    try:
        page = ctx.new_page()
        page.goto(url, wait_until="domcontentloaded", timeout=30_000)
        page.wait_for_load_state("networkidle", timeout=30_000)

        import time
        time.sleep(2)  # settle for lazy-load

        scroll_lazy(page)
        dismiss_cookies(page, club=club)

        if hide_selectors:
            css = "".join(f"{sel}{{display:none !important}}" for sel in hide_selectors)
            page.add_style_tag(content=css)

        png_bytes = page.screenshot(full_page=True)
        out_file.write_bytes(png_bytes)
        logger.info(f"Captured {club} {step_name} → {out_file} ({len(png_bytes)} bytes)")
        return out_file
    finally:
        ctx.close()
        pw.stop()
```
